API 使用文档

所有接口通过 X-Api-Key Header 认证。

认证与占位符

认证 Header

X-Api-Key: djd_YOUR_API_KEY

示例变量

BASE_URL={{ base_url }}
BUCKET_ID=<bucket_id>
FILE_ID=<file_id>

创建 API Key 后只会展示一次完整密钥，请及时保存。示例中的 djd_YOUR_API_KEY 需要替换为真实密钥。

存储桶接口速查

操作 Method Endpoint 请求数据
列出桶 GET {{ base_url }}/buckets/api/buckets/ 无
创建桶 POST {{ base_url }}/buckets/api/buckets/create/ {"name":"photos","is_public":false}
删除桶 DELETE {{ base_url }}/buckets/api/buckets/<bucket_id>/ 无，删除后不可恢复
列出桶内文件 GET {{ base_url }}/buckets/api/buckets/<bucket_id>/files/?path=docs/ path 可选，目录以 / 结尾
上传文件 POST {{ base_url }}/buckets/api/buckets/<bucket_id>/upload/ multipart: files 一个或多个，folder_path 可选
下载桶文件 GET {{ base_url }}/buckets/api/<bucket_id>/files/<file_id>/download/ 返回文件流
删除桶文件 DELETE {{ base_url }}/buckets/api/buckets/<bucket_id>/files/<file_id>/ 无
创建目录 POST {{ base_url }}/buckets/api/buckets/<bucket_id>/folder/create/ {"name":"docs","parent_path":""}
删除目录 DELETE {{ base_url }}/buckets/api/buckets/<bucket_id>/folder/?path=docs/ 删除目录下所有文件

多语言常用操作样例

每种语言都包含：列出桶、创建桶、删除桶、列出文件、上传文件、下载文件、创建目录、删除文件、删除目录。

cURL

BASE_URL="{{ base_url }}"
API_KEY="djd_YOUR_API_KEY"
BUCKET_ID="<bucket_id>"
FILE_ID="<file_id>"

# 列出桶
curl -H "X-Api-Key: $API_KEY" "$BASE_URL/buckets/api/buckets/"

# 创建桶
curl -X POST "$BASE_URL/buckets/api/buckets/create/" \
  -H "X-Api-Key: $API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"name":"photos","is_public":false}'

# 删除桶
curl -X DELETE -H "X-Api-Key: $API_KEY" \
  "$BASE_URL/buckets/api/buckets/$BUCKET_ID/"

# 列出文件
curl -H "X-Api-Key: $API_KEY" \
  "$BASE_URL/buckets/api/buckets/$BUCKET_ID/files/?path=docs/"

# 上传文件
curl -X POST "$BASE_URL/buckets/api/buckets/$BUCKET_ID/upload/" \
  -H "X-Api-Key: $API_KEY" \
  -F "files=@./readme.md" \
  -F "folder_path=docs/"

# 下载文件
curl -H "X-Api-Key: $API_KEY" -o readme.md \
  "$BASE_URL/buckets/api/$BUCKET_ID/files/$FILE_ID/download/"

# 创建目录
curl -X POST "$BASE_URL/buckets/api/buckets/$BUCKET_ID/folder/create/" \
  -H "X-Api-Key: $API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"name":"docs","parent_path":""}'

# 删除文件
curl -X DELETE -H "X-Api-Key: $API_KEY" \
  "$BASE_URL/buckets/api/buckets/$BUCKET_ID/files/$FILE_ID/"

# 删除目录
curl -X DELETE -H "X-Api-Key: $API_KEY" \
  "$BASE_URL/buckets/api/buckets/$BUCKET_ID/folder/?path=docs/"

JavaScript / TypeScript

const BASE_URL = "{{ base_url }}";
const API_KEY = "djd_YOUR_API_KEY";
const BUCKET_ID = "<bucket_id>";
const FILE_ID = "<file_id>";

const headers = { "X-Api-Key": API_KEY };
const jsonHeaders = { ...headers, "Content-Type": "application/json" };

async function listBuckets() {
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/`, { headers });
  return res.json();
}

async function createBucket() {
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/create/`, {
    method: "POST",
    headers: jsonHeaders,
    body: JSON.stringify({ name: "photos", is_public: false })
  });
  return res.json();
}

async function deleteBucket() {
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/`, {
    method: "DELETE",
    headers
  });
  return res.json();
}

async function listFiles(path = "docs/") {
  const url = new URL(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/files/`);
  url.searchParams.set("path", path);
  const res = await fetch(url, { headers });
  return res.json();
}

async function uploadFile(file) {
  const body = new FormData();
  body.append("files", file);
  body.append("folder_path", "docs/");
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/upload/`, {
    method: "POST",
    headers,
    body
  });
  return res.json();
}

async function downloadFile() {
  const res = await fetch(`${BASE_URL}/buckets/api/${BUCKET_ID}/files/${FILE_ID}/download/`, { headers });
  const blob = await res.blob();
  return blob;
}

async function createFolder() {
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/folder/create/`, {
    method: "POST",
    headers: jsonHeaders,
    body: JSON.stringify({ name: "docs", parent_path: "" })
  });
  return res.json();
}

async function deleteFile() {
  const res = await fetch(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/files/${FILE_ID}/`, {
    method: "DELETE",
    headers
  });
  return res.json();
}

async function deleteFolder(path = "docs/") {
  const url = new URL(`${BASE_URL}/buckets/api/buckets/${BUCKET_ID}/folder/`);
  url.searchParams.set("path", path);
  const res = await fetch(url, { method: "DELETE", headers });
  return res.json();
}

import requests

BASE_URL = "{{ base_url }}"
API_KEY = "djd_YOUR_API_KEY"
BUCKET_ID = "<bucket_id>"
FILE_ID = "<file_id>"
HEADERS = {"X-Api-Key": API_KEY}

def list_buckets():
    return requests.get(f"{BASE_URL}/buckets/api/buckets/", headers=HEADERS).json()

def create_bucket():
    payload = {"name": "photos", "is_public": False}
    return requests.post(f"{BASE_URL}/buckets/api/buckets/create/", json=payload, headers=HEADERS).json()

def delete_bucket():
    return requests.delete(f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/", headers=HEADERS).json()

def list_files(path="docs/"):
    return requests.get(
        f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/files/",
        params={"path": path},
        headers=HEADERS,
    ).json()

def upload_file():
    with open("readme.md", "rb") as f:
        files = [("files", ("readme.md", f, "text/markdown"))]
        data = {"folder_path": "docs/"}
        return requests.post(
            f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/upload/",
            headers=HEADERS,
            files=files,
            data=data,
        ).json()

def download_file():
    r = requests.get(f"{BASE_URL}/buckets/api/{BUCKET_ID}/files/{FILE_ID}/download/", headers=HEADERS)
    r.raise_for_status()
    open("readme.md", "wb").write(r.content)

def create_folder():
    payload = {"name": "docs", "parent_path": ""}
    return requests.post(
        f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/folder/create/",
        json=payload,
        headers=HEADERS,
    ).json()

def delete_file():
    return requests.delete(f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/files/{FILE_ID}/", headers=HEADERS).json()

def delete_folder(path="docs/"):
    return requests.delete(
        f"{BASE_URL}/buckets/api/buckets/{BUCKET_ID}/folder/",
        params={"path": path},
        headers=HEADERS,
    ).json()

package main

import (
	"bytes"
	"io"
	"mime/multipart"
	"net/http"
	"net/url"
	"os"
)

const baseURL = "{{ base_url }}"
const apiKey = "djd_YOUR_API_KEY"
const bucketID = "<bucket_id>"
const fileID = "<file_id>"

func do(method, endpoint, contentType string, body io.Reader) (*http.Response, error) {
	req, err := http.NewRequest(method, baseURL+endpoint, body)
	if err != nil {
		return nil, err
	}
	req.Header.Set("X-Api-Key", apiKey)
	if contentType != "" {
		req.Header.Set("Content-Type", contentType)
	}
	return http.DefaultClient.Do(req)
}

func listBuckets() (*http.Response, error) {
	return do("GET", "/buckets/api/buckets/", "", nil)
}

func createBucket() (*http.Response, error) {
	body := bytes.NewBufferString(`{"name":"photos","is_public":false}`)
	return do("POST", "/buckets/api/buckets/create/", "application/json", body)
}

func deleteBucket() (*http.Response, error) {
	return do("DELETE", "/buckets/api/buckets/"+bucketID+"/", "", nil)
}

func listFiles(path string) (*http.Response, error) {
	q := url.Values{}
	q.Set("path", path)
	return do("GET", "/buckets/api/buckets/"+bucketID+"/files/?"+q.Encode(), "", nil)
}

func uploadFile() (*http.Response, error) {
	var body bytes.Buffer
	writer := multipart.NewWriter(&body)
	writer.WriteField("folder_path", "docs/")
	part, err := writer.CreateFormFile("files", "readme.md")
	if err != nil {
		return nil, err
	}
	file, err := os.Open("readme.md")
	if err != nil {
		return nil, err
	}
	defer file.Close()
	io.Copy(part, file)
	writer.Close()
	return do("POST", "/buckets/api/buckets/"+bucketID+"/upload/", writer.FormDataContentType(), &body)
}

func downloadFile() (*http.Response, error) {
	return do("GET", "/buckets/api/"+bucketID+"/files/"+fileID+"/download/", "", nil)
}

func createFolder() (*http.Response, error) {
	body := bytes.NewBufferString(`{"name":"docs","parent_path":""}`)
	return do("POST", "/buckets/api/buckets/"+bucketID+"/folder/create/", "application/json", body)
}

func deleteFile() (*http.Response, error) {
	return do("DELETE", "/buckets/api/buckets/"+bucketID+"/files/"+fileID+"/", "", nil)
}

func deleteFolder(path string) (*http.Response, error) {
	q := url.Values{}
	q.Set("path", path)
	return do("DELETE", "/buckets/api/buckets/"+bucketID+"/folder/?"+q.Encode(), "", nil)
}

function request_api($method, $url, $body = null, $headers = []) {
    global $apiKey;
    $ch = curl_init($url);

    curl_setopt_array($ch, [

Java 11+

import java.io.IOException;
import java.net.URI;
import java.net.URLEncoder;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;

public class BucketApiExample {
    static final String BASE_URL = "{{ base_url }}";
    static final String API_KEY = "djd_YOUR_API_KEY";
    static final String BUCKET_ID = "<bucket_id>";
    static final String FILE_ID = "<file_id>";
    static final HttpClient CLIENT = HttpClient.newHttpClient();

    static HttpRequest.Builder request(String path) {
        return HttpRequest.newBuilder(URI.create(BASE_URL + path))
                .header("X-Api-Key", API_KEY);
    }

    static HttpResponse<String> send(HttpRequest request) throws Exception {
        return CLIENT.send(request, HttpResponse.BodyHandlers.ofString());
    }

    static HttpResponse<String> listBuckets() throws Exception {
        return send(request("/buckets/api/buckets/").GET().build());
    }

    static HttpResponse<String> createBucket() throws Exception {
        String json = "{\"name\":\"photos\",\"is_public\":false}";
        return send(request("/buckets/api/buckets/create/")
                .header("Content-Type", "application/json")
                .POST(HttpRequest.BodyPublishers.ofString(json))
                .build());
    }

    static HttpResponse<String> deleteBucket() throws Exception {
        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/")
                .DELETE()
                .build());
    }

    static HttpResponse<String> listFiles(String path) throws Exception {
        String q = URLEncoder.encode(path, StandardCharsets.UTF_8);
        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/files/?path=" + q).GET().build());
    }

    static HttpResponse<String> uploadFile() throws Exception {
        String boundary = "----DjangoDiskBoundary";
        byte[] fileBytes = Files.readAllBytes(Path.of("readme.md"));
        String prefix = "--" + boundary + "\r\n"
                + "Content-Disposition: form-data; name=\"folder_path\"\r\n\r\ndocs/\r\n"
                + "--" + boundary + "\r\n"
                + "Content-Disposition: form-data; name=\"files\"; filename=\"readme.md\"\r\n"
                + "Content-Type: text/markdown\r\n\r\n";
        String suffix = "\r\n--" + boundary + "--\r\n";
        byte[] body = concat(prefix.getBytes(StandardCharsets.UTF_8), fileBytes, suffix.getBytes(StandardCharsets.UTF_8));

        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/upload/")
                .header("Content-Type", "multipart/form-data; boundary=" + boundary)
                .POST(HttpRequest.BodyPublishers.ofByteArray(body))
                .build());
    }

    static HttpResponse<byte[]> downloadFile() throws Exception {
        return CLIENT.send(
                request("/buckets/api/" + BUCKET_ID + "/files/" + FILE_ID + "/download/").GET().build(),
                HttpResponse.BodyHandlers.ofByteArray());
    }

    static HttpResponse<String> createFolder() throws Exception {
        String json = "{\"name\":\"docs\",\"parent_path\":\"\"}";
        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/folder/create/")
                .header("Content-Type", "application/json")
                .POST(HttpRequest.BodyPublishers.ofString(json))
                .build());
    }

    static HttpResponse<String> deleteFile() throws Exception {
        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/files/" + FILE_ID + "/").DELETE().build());
    }

    static HttpResponse<String> deleteFolder(String path) throws Exception {
        String q = URLEncoder.encode(path, StandardCharsets.UTF_8);
        return send(request("/buckets/api/buckets/" + BUCKET_ID + "/folder/?path=" + q).DELETE().build());
    }

    static byte[] concat(byte[]... arrays) throws IOException {
        int len = 0;
        for (byte[] a : arrays) len += a.length;
        byte[] out = new byte[len];
        int pos = 0;
        for (byte[] a : arrays) {
            System.arraycopy(a, 0, out, pos, a.length);
            pos += a.length;
        }
        return out;
    }
}

using System.Net.Http.Headers;
using System.Text;
using System.Text.Json;

var baseUrl = "{{ base_url }}";
var apiKey = "djd_YOUR_API_KEY";
var bucketId = "<bucket_id>";
var fileId = "<file_id>";

using var client = new HttpClient();
client.DefaultRequestHeaders.Add("X-Api-Key", apiKey);

async Task<string> ListBuckets() {
    return await client.GetStringAsync($"{baseUrl}/buckets/api/buckets/");
}

async Task<string> CreateBucket() {
    var json = JsonSerializer.Serialize(new { name = "photos", is_public = false });
    var res = await client.PostAsync($"{baseUrl}/buckets/api/buckets/create/",
        new StringContent(json, Encoding.UTF8, "application/json"));
    return await res.Content.ReadAsStringAsync();
}

async Task<string> DeleteBucket() {
    var res = await client.DeleteAsync($"{baseUrl}/buckets/api/buckets/{bucketId}/");
    return await res.Content.ReadAsStringAsync();
}

async Task<string> ListFiles(string path = "docs/") {
    var url = $"{baseUrl}/buckets/api/buckets/{bucketId}/files/?path={Uri.EscapeDataString(path)}";
    return await client.GetStringAsync(url);
}

async Task<string> UploadFile() {
    using var form = new MultipartFormDataContent();
    form.Add(new StringContent("docs/"), "folder_path");
    var file = new StreamContent(File.OpenRead("readme.md"));
    file.Headers.ContentType = new MediaTypeHeaderValue("text/markdown");
    form.Add(file, "files", "readme.md");
    var res = await client.PostAsync($"{baseUrl}/buckets/api/buckets/{bucketId}/upload/", form);
    return await res.Content.ReadAsStringAsync();
}

async Task DownloadFile() {
    var bytes = await client.GetByteArrayAsync($"{baseUrl}/buckets/api/{bucketId}/files/{fileId}/download/");
    await File.WriteAllBytesAsync("readme.md", bytes);
}

async Task<string> CreateFolder() {
    var json = JsonSerializer.Serialize(new { name = "docs", parent_path = "" });
    var res = await client.PostAsync($"{baseUrl}/buckets/api/buckets/{bucketId}/folder/create/",
        new StringContent(json, Encoding.UTF8, "application/json"));
    return await res.Content.ReadAsStringAsync();
}

async Task<string> DeleteFile() {
    var res = await client.DeleteAsync($"{baseUrl}/buckets/api/buckets/{bucketId}/files/{fileId}/");
    return await res.Content.ReadAsStringAsync();
}

async Task<string> DeleteFolder(string path = "docs/") {
    var url = $"{baseUrl}/buckets/api/buckets/{bucketId}/folder/?path={Uri.EscapeDataString(path)}";
    var res = await client.DeleteAsync(url);
    return await res.Content.ReadAsStringAsync();
}

def request(method, path, body: nil, content_type: nil)
  uri = URI(BASE_URL + path)
  req = Object.const_get("Net::HTTP::#{method.capitalize}").new(uri)
  req["X-Api-Key"] = API_KEY
  req["Content-Type"] = content_type if content_type

end
